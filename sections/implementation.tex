 \begin{frame}
	 \frametitle{Installation}
        \framesubtitle{\mbox{}}
        \centering
	\begin{enumerate}
		\item \texttt{git clone git@github.com:schemaanalyst-team/schemaanalyst.git}
		\item \texttt{./gradlew compile}
		\item \texttt{export CLASSPATH="build/classes/main:lib/*:build/lib/*:."}
	\end{enumerate}
 \end{frame}
